
------------------------------------------------
▼アニメーション終了時にエフェクトオブジェクトを消去する
　　AnimationFinished.csを作って以下をコピペ
　　↓
　　アニメーションウィンドウのTLの最後にこのメソッドを指定
------------------------------------------------
public class AnimationFinished : MonoBehaviour
{
    // -----------------------
    // メカニムのアニメーションイベントからコールされる
    // エフェクトを破棄
    // -----------------------
    void OnAnimationFinish()
    {
        Destroy(this.gameObject);
    }
}
